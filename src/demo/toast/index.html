Usage
AngularJS

引入服务$ui.toast

angular .module( "app", [ "$ui.toast" ] ) .controller( "toastController", [ "$toast", function( $toast ) { $toast.top( "On the top left!" ).left(); } ] );
jQuery

引入模块ui/toast/toast

require( [ "ui/toast/toast" ], function() { $.toast.top( "On the top left!" ).left(); } );
Method

top( message, [ delay | class4toast ], [ class4toast ] ).left(); - 左上角显示

message - String, 文本内容

delay - Number, 自动关闭延时, 单位毫秒

class4toast - String, 主题名称p>



top( message, [ delay | class4toast ], [ class4toast ] ).right(); - 右上角显示

message - String, 文本内容

delay - Number, 自动关闭延时, 单位毫秒

class4toast - String, 主题名称p>



bottom( message, [ delay | class4toast ], [ class4toast ] ).left(); - 左下角显示

message - String, 文本内容

delay - Number, 自动关闭延时, 单位毫秒

class4toast - String, 主题名称p>



bottom( message, [ delay | class4toast ], [ class4toast ] ).right(); - 右下角显示

message - String, 文本内容

delay - Number, 自动关闭延时, 单位毫秒

class4toast - String, 主题名称p>
